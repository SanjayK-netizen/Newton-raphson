function calculate() { let fxInput = document.getElementById("fx").value; let x0 = parseFloat(document.getElementById("x0").value); let maxIter = parseInt(document.getElementById("iter").value); let tol = parseFloat(document.getElementById("tol").value); // Parse function using math.js let f = math.parse(fxInput).compile(); // Compute derivative automatically let derivative = math.derivative(fxInput, 'x').compile(); let table = document.getElementById("table"); table.innerHTML = ` Iteration xₙ f(xₙ) xₙ₊₁ Error `; let xn = x0; for (let i = 1; i <= maxIter; i++) { let fxn = f.evaluate({x: xn}); let dfxn = derivative.evaluate({x: xn}); if (dfxn === 0) { document.getElementById("result").innerText = "Derivative is zero. Method fails."; return; } let xn1 = xn - fxn / dfxn; let error = Math.abs(xn1 - xn); let row = table.insertRow(); row.insertCell(0).innerText = i; row.insertCell(1).innerText = xn.toFixed(6); row.insertCell(2).innerText = fxn.toFixed(6); row.insertCell(3).innerText = xn1.toFixed(6); row.insertCell(4).innerText = error.toFixed(6); if (error < tol) { document.getElementById("result").innerText = `Root found: ${xn1.toFixed(6)}`; return; } xn = xn1; } document.getElementById("result").innerText = `Approximate root after ${maxIter} iterations: ${xn.toFixed(6)}`; } function reset() { document.getElementById("fx").value = ""; document.getElementById("x0").value = ""; document.getElementById("iter").value = 10; document.getElementById("tol").value = 0.0001; document.getElementById("result").innerText = ""; document.getElementById("table").innerHTML = ` Iteration xₙ f(xₙ) xₙ₊₁ Error `; }
